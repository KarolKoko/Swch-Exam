Program 1

Sprawdź, czy zbiór liczb zawiera daną liczbę. Program wykorzystuje algorytm wyszukiwania binarnego.

JavaScript 

const numbersArrayValidator = (array) => {
    if (Array.isArray(array)) {
        if (array.length > 0) {
        let x = 0;
        while (x < array.length) {
            if (isNaN(array[x])) {
            break;
            } else {
            x = x + 1;
            }
        }
        if (x == array.length) {
            return true;
        } else {
            return false;
        }
        }
    } else {
        return false;
    }
};
    
const doesArrayContainsNumber = (number, array) => {
    if (!isNaN(number) && numbersArrayValidator(array)) {
        const sortedNumbersArray = array.sort((a, b) => a - b);
        let firstIndex = 0;
        let lastIndex = sortedNumbersArray.length - 1;
        let currentIndex;
        while (firstIndex <= lastIndex) {
        currentIndex = Math.floor((firstIndex + lastIndex) / 2);
        if (sortedNumbersArray[currentIndex] < number) {
            firstIndex = currentIndex + 1;
        } else if (sortedNumbersArray[currentIndex] > number) {
            lastIndex = currentIndex - 1;
        } else {
            return 'Tablica zawiera podaną liczbę';
        }
        }
        return 'Tablica nie zawiera podanej liczby';
    } else {
        return 'Error 400 - błędne zapytanie';
    }
};
